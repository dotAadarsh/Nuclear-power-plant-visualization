import folium import pandas as pd import json from pyodide.http import open_url location = "./World_Nuclear_Power_Plant.csv" plant_station_locations = pd.read_csv(open_url(location)) plant_station_locations = plant_station_locations[["latitude", "longitude", "powerplant"]] map = folium.Map(location=[plant_station_locations.latitude.mean(), plant_station_locations.longitude.mean()], tiles="cartodbdark_matter", zoom_start=5,control_scale=True) for index, location_info in plant_station_locations.iterrows(): folium.Marker([location_info["latitude"], location_info["longitude"]], popup=location_info["powerplant"]).add_to(map) map
